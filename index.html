Fizz-Buzz

Ruby

            
              0.upto(99) do |i|
                if i % 5 == 0 && i % 3 ==0
                  puts "Fizz-Buzz"
                elsif i % 3 == 0
                  puts "Fizz"
                elsif i % 5 == 0
                  puts "Buzz"
                end
              end
